program AdvancedTest;

const
  phi = 3.14;
  max_speed = 100;

type
  Colors = (Red, Green, Blue);
  Point = record
    x, y: integer;
  end;

var
  p1: Point;
  c: Colors;
  i: integer;

procedure MovePoint(dx, dy: integer);
begin
  p1.x := p1.x + dx;
  p1.y := p1.y + dy;
end;

begin
  p1.x := 0;
  p1.y := 0;
  c := Red;

  for i := 1 to 5 do
    MovePoint(i, i * 2);

  case i of
    1, 2: writeln('Low');
    3, 4: writeln('Medium');
    5: writeln('High');
  end;
end.
